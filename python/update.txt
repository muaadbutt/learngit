update inventtransferline 
set UNDERDELIVERYPCT = '100', 
OVERDELIVERYPCT         = '100'
from inventtransferline itl_upd
join (
select distinct tll.INVENTTRANSFERLINERECID
--itt.TRANSFERSTATUS InventTransferTable_Status,tbl.ISSYNCRONIZE,tbl.TRANSFERID,tll.INVENTTRANSFERLINERECID
--,tbl.TRANSFERSTATUS,tll.RETAILVARIANTID,tll.QTYSHIPPED,itl.QTYRECEIVED,itl.QTYREMAINRECEIVE,itl.QTYRECEIVENOW,tbl.CARTONNO,tbl.RECID,tbl.CREATEDDATETIME
from VSI_TRANSFERORDERLOGTABLE tbl
inner join VSI_TRANSFERORDERLOGLINES tll on tbl.RECID = tll.REFRECID
inner join INVENTTRANSFERLINE itl on itl.RECID = tll.INVENTTRANSFERLINERECID
inner join INVENTTRANSFERTABLE itt on itt.TRANSFERID = itl.TRANSFERID
where ISSYNCRONIZE =0  and itl.UNDERDELIVERYPCT = 0 
) a on A.INVENTTRANSFERLINERECID =itl_upd.RECID
